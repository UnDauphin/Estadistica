Bucle para variar n

for (n in n_values) {
  # Inicializar el vector z para almacenar los estadÃ­sticos
  z <- rep(NA, M)
  
  # Bucle para generar muestras y calcular estadÃ­sticos
  for (i in 1:M) {
    sample_ages <- sample(df_filtered$Age, n)
    sample_mean <- mean(sample_ages)
    sample_sd <- sd(sample_ages)
    z[i] <- (sample_mean - mean(df_filtered$Age)) / (sample_sd / sqrt(n))

  # Prueba de normalidad
  print(lillie.test(z))
